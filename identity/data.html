User Data


                  <?php

                  namespace Application\Controllers;

                  use DevNet\Core\Controller\AbstractController;
                  use DevNet\Core\Controller\IActionResult;
                  use DevNet\Core\Identity\UserManager;
                  use DevNet\Core\Identity\User;
                  use Application\Models\RegisterForm;

                  class AccountController extends AbstractController
                  {
                     private UserManager $Users;
                     
                     public function __construct(UserManager $users)
                     {
                        $this->Users = $users;
                     }
                     ...
                  }
               

                  public function create(RegisterForm $form) : IActionResult
                  {
                     // if one or more fields are empty then show registration page.
                     if (!$form->isValide())
                     {
                           return $this->view('account/register');
                     }

                     $user = new User();
                     $user->Username = $form->Email;
                     $user->Password = $form->Password;
                     $this->Users->create($user);

                     return $this->content("User {$user->Username} has been created");
                  }
               

                  public function update() : IActionResult
                  {
                     $user = $this->Users->getUser();
                     $user->Password = "NewPassword";
                     $this->Users->update($user);

                     return $this->redirect('home/index');
                  }
               

                  public function delete() : IActionResult
                  {
                     $user = $this->Users->getUser();
                     $this->Users->delete($user);

                     return $this->redirect('home/index');
                  }
